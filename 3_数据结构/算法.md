```python
def BubbleSort(a):
   n=len(a)
   for i in range(n):
       for j in range (n-i-1):
           if a[j+1]<a[j]:
               a[j+1],a[j]= a[j],a[j+1]
   return a
```

#### 2.选择排序
```python
def SelectSort(a):
    n=len(a)
    for i in range(n):
        min_indx=i
        for j in range(i+1,n):
            if a[j] <a[min_indx]:
                a[j],a[min_indx]=a[min_indx],a[j]
    return a
```

#### 3.快速排序
```python
def QuickSort(a):
    n=len(a)
    if n <= 1:
        return a
    Base_a=a[n//2]
    left=[a[x] for x in range(n) if a[x] < Base_a]
    middle=[a[x] for x in range(n) if a[x] ==Base_a]
    right=[a[x] for x in range(n) if a[x]> Base_a]
    return  QuickSort(left)+middle+QuickSort(right)

```

#### 4.二分查找

```python
def BinarySearch(a, bottom, top, x):
    if top >= bottom:
        middle = bottom + (top - bottom ) // 2
        if x == a[middle]:
            return a[middle]
        elif x < a[middle]:
            return BinarySearch(a, bottom, middle - 1, x)
        else:
            return BinarySearch(a, middle + 1, top, x)
    else:

        print("元素不存在")
        return -1
```

```python
#用装饰器实现单例模式
def singleton(cls,*args,**kwargs):
    instance={}
    def wrapper():
        if cls not in instance:
            instance[cls]=cls(*args,**kwargs)
        return instance[cls]
    return wrapper

@singleton
class test_singleton(object):
    def __init__(self):
        self.num_sum=0
```
